// aabb와 frustum 컬링
bool CFrustum::IsCulled( const CAABB& aabb )
{
int nTotalIn = 0;

const D3DXVECTOR3* corners = aabb.GetCorners();
for (int p=0; p < FRUSTUM_PLANE_COUNT; ++p)
{
int nInCount = 8;
int nPointIn = 1;
for (int i=0; i < 8; ++i)
{
if (m_aPlane[p].GetSide( corners[i] ) == CPlane::NEGATIVE_SIDE)
{
nPointIn = 0;
--nInCount;
}
}
if (nInCount == 0)
return true;

nTotalIn += nInCount;
}

// inside
if (nTotalIn == FRUSTUM_PLANE_COUNT)
return false;
return false;
}
