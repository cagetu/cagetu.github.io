3) in-place 변수 -> pointer 변수 + 전방선언 (가독성이 떨어짐으로 간절히 필요한 곳(desperate)에만 사용)
// a.h
class B;
class A 
{ 
 A();
 B *_b;
 static const int SIZE_OF_B = 20;
 char _b_storage[SIZE_OF_B];
}; 

// a.cpp 
#include ”b.h” 
A::A()
{
 XASSERT(sizeof(B) == SIZE_OF_B);
 _b = new (_b_storage) B(); 
}
