// 디폴트 패스

// 실루엣 패스 
String shaderDef = TEXT("_USE_DIFFUSEMAP,_USE_NORMAL,_USE_TEXCOORD,_SKINNING,_SILHOUETTE");
uint32 shadermask = ShaderFeature::GetShaderMask(shaderDef);
ptr<Shader> shader = ShaderCore::Build(material->GetShaderCode(), shadermask);

pass = material->AddPass(shader);
if (pass)
{
pass->SetValue(TEXT("SILHOUETTEPARAM"), Vector4(0.0f, 0.0f, 0.0f, 6.0f));
pass->SetRenderState(CULL_CW);
}
