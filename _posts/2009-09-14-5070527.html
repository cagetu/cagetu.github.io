하지만, 모든 것에 같은 깊이버퍼를 요구하는 것은 아니다.

뷰 공간 상의 Depth 인지 동차좌표계 상의 Depth 인지, 잘 판단해서 적절하게 필요한 곳에 사용하도록 하자!!


//------------------------------------------------------------------------------
/**
    EncodeDepth()
   
    Encode depth into 2 8-bit values.
*/
float2
EncodeDepth(float d)
{
    return float2(floor(d) / 256.0, frac(d));
}

//------------------------------------------------------------------------------
/**
   DecodeDepth()
   
    Decode depth from 2 8-bit values.
*/
float
DecodeDepth(float2 codedDepth)
{
    return codedDepth.x * 256.0 + codedDepth.y;
}


//------------------------------------------------------------------------------
/**
    Vertex shader function for the depth pass.
    Passes depth to pixelshader.   
*/
vsDepthPassOutput
DepthPassVertexShader(float4 pos : POSITION)
{
    vsDepthPassOutput vsOut;
    vsOut.position = mul(pos, mvp);
    vsOut.depth.x  = vsOut.position.z;
    return vsOut;
}
//------------------------------------------------------------------------------
/**
*/
float4 DepthPassPixelShader(const vsDepthPassOutput psIn) : COLOR
{
    return float4(EncodeDepth(psIn.depth.x), 0.0f, 1.0f);
}
